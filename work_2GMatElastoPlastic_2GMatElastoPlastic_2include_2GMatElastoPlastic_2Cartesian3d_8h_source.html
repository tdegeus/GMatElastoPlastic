111 void refresh(bool compute_tangent = true) override
112 {
113 namespace GT = GMatTensor::Cartesian3d::pointer;
114
115#pragma omp parallel
116 {
117 auto I = GMatTensor::Cartesian3d::I2();
118 auto II = GMatTensor::Cartesian3d::II();
119 auto I4d = GMatTensor::Cartesian3d::I4d();
120
121 double K;
122 double G;
123 double sigy0;
124 double H;
125 double epsp_t;
126
127 auto Eps = xt::adapt(m_Eps.data(), {m_ndim, m_ndim});
128 auto Eps_t = xt::adapt(m_Eps_t.data(), {m_ndim, m_ndim});
129 auto Epse = xt::adapt(m_Epse.data(), {m_ndim, m_ndim});
130 auto Epse_t = xt::adapt(m_Epse_t.data(), {m_ndim, m_ndim});
131 auto Sig = xt::adapt(m_Sig.data(), {m_ndim, m_ndim});
132 auto C = xt::adapt(m_C.data(), {m_ndim, m_ndim, m_ndim, m_ndim});
133
134 auto Epsed = xt::empty_like(I);
135
136#pragma omp for
137 for (size_t i = 0; i < m_size; ++i) {
138
139 K = m_K.flat(i);
140 G = m_G.flat(i);
141 sigy0 = m_sigy0.flat(i);
142 H = m_H.flat(i);
143 epsp_t = m_epsp_t.flat(i);
144
145 Eps.reset_buffer(&m_Eps.flat(i * m_stride_tensor2), m_stride_tensor2);
146 Eps_t.reset_buffer(&m_Eps_t.flat(i * m_stride_tensor2), m_stride_tensor2);
147 Epse.reset_buffer(&m_Epse.flat(i * m_stride_tensor2), m_stride_tensor2);
148 Epse_t.reset_buffer(&m_Epse_t.flat(i * m_stride_tensor2), m_stride_tensor2);
149 Sig.reset_buffer(&m_Sig.flat(i * m_stride_tensor2), m_stride_tensor2);
150 C.reset_buffer(&m_C.flat(i * m_stride_tensor4), m_stride_tensor4);
151
152 // trial elastic strain: presume the strain increment to be fully elastic
153 xt::noalias(Epse) = Epse_t + (Eps - Eps_t);
154
155 // decompose trial elastic strain
156 double epsem = GT::Hydrostatic_deviatoric(Epse.data(), Epsed.data());
157
158 // trial stress
159 double sigm = 3.0 * K * epsem;
160 auto Sigd = xt::eval(2.0 * G * Epsed);
161 double sigeq = std::sqrt(1.5 * GT::A2s_ddot_B2s(Sigd.data(), Sigd.data()));
162
163 // evaluate the yield surface
164 double phi = sigeq - (sigy0 + H * epsp_t);
165
166 // elastic tangent
167 if (compute_tangent) {
168 xt::noalias(C) = K * II + 2.0 * G * I4d;
169 }
170
171 // return map
172 if (phi > 0) {
173 auto N2 = xt::eval(1.5 * Sigd / sigeq);
174 double dgamma = phi / (3.0 * G + H);
175 Sigd *= (1.0 - 3.0 * G * dgamma / sigeq);
176 xt::noalias(Epsed) = Sigd / (2.0 * G);
177 xt::noalias(Epse) = epsem * I + Epsed;
178 m_epsp.flat(i) = epsp_t + dgamma;
179 if (compute_tangent) {
180 auto NN = xt::empty_like(II);
181 GT::A2_dyadic_B2(N2.data(), N2.data(), NN.data());
182 C -= 6.0 * std::pow(G, 2.0) * dgamma / sigeq * I4d;
183 C += 4.0 * std::pow(G, 2.0) * (dgamma / sigeq - 1.0 / (3.0 * G + H)) * NN;
184 }
185 }
186
187 xt::noalias(Sig) = sigm * I + Sigd;
188 }
189 }
190 }
