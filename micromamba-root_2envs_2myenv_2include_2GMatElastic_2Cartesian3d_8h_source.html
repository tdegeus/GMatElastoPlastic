123 template <class T>
124 void init_Elastic(const T& K, const T& G)
125 {
126 GMATELASTIC_ASSERT(K.dimension() == N);
127 GMATELASTIC_ASSERT(xt::has_shape(K, G.shape()));
128 std::copy(K.shape().cbegin(), K.shape().cend(), m_shape.begin());
129 this->init(m_shape);
130
131 m_K = K;
132 m_G = G;
133 m_Eps = xt::zeros<double>(m_shape_tensor2);
134 m_Sig = xt::zeros<double>(m_shape_tensor2);
135 m_C = xt::empty<double>(m_shape_tensor4);
136
137#pragma omp parallel
138 {
139 auto C = xt::adapt(m_C.data(), {m_ndim, m_ndim, m_ndim, m_ndim});
140 double K;
141 double G;

143 auto I4d = GMatTensor::Cartesian3d::I4d();
144
145#pragma omp for
146 for (size_t i = 0; i < m_size; ++i) {
147 C.reset_buffer(&m_C.flat(i * m_stride_tensor4), m_stride_tensor4);
148 K = m_K.flat(i);
149 G = m_G.flat(i);
150 C = K * II + 2.0 * G * I4d;

152 }

224 virtual void refresh(bool compute_tangent = true)
225 {
226 (void)(compute_tangent);
227
228#pragma omp parallel for
229 for (size_t i = 0; i < m_size; ++i) {
230
231 double K = m_K.flat(i);
232 double G = m_G.flat(i);
233
234 const double* Eps = &m_Eps.flat(i * m_stride_tensor2);
235 double* Sig = &m_Sig.flat(i * m_stride_tensor2);
236
237 double epsm = GMatTensor::Cartesian3d::pointer::Hydrostatic(Eps);
238
239 Sig[0] = (3.0 * K - 2.0 * G) * epsm + 2.0 * G * Eps[0];
240 Sig[1] = 2.0 * G * Eps[1];
241 Sig[2] = 2.0 * G * Eps[2];
242 Sig[3] = 2.0 * G * Eps[3];
243 Sig[4] = (3.0 * K - 2.0 * G) * epsm + 2.0 * G * Eps[4];
244 Sig[5] = 2.0 * G * Eps[5];
245 Sig[6] = 2.0 * G * Eps[6];
246 Sig[7] = 2.0 * G * Eps[7];
247 Sig[8] = (3.0 * K - 2.0 * G) * epsm + 2.0 * G * Eps[8];
248 }

292 virtual array_type::tensor<double, N> energy() const
293 {
294 array_type::tensor<double, N> ret = xt::empty<double>(m_shape);
295 namespace GT = GMatTensor::Cartesian3d::pointer;
296
297#pragma omp parallel for
298 for (size_t i = 0; i < m_size; ++i) {
299
300 double K = m_K.flat(i);
301 double G = m_G.flat(i);
302
303 const double* Eps = &m_Eps.flat(i * m_stride_tensor2);
304
305 std::array<double, m_stride_tensor2> Epsd;
306 double epsm = GT::Hydrostatic_deviatoric(Eps, &Epsd[0]);
307 double epsd = std::sqrt(0.5 * GT::A2s_ddot_B2s(&Epsd[0], &Epsd[0]));
308
309 double U = 3.0 * K * std::pow(epsm, 2.0);
310 double V = 2.0 * G * std::pow(epsd, 2.0);
311
312 ret.flat(i) = U + V;
313 }
314
315 return ret;
316 }
